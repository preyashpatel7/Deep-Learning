# Calculating change in weights and bias from second hidden layer to the output layer
    l3_error = out_y - y
    dW3 = np.dot(out_h2.T, l3_error)
    db3 = np.sum(l3_error, axis = 0, keepdims=True)

    # Calculating change in weights and bias from first hidden layer to the second hidden layer
    dh2 = np.dot(w3, l3_error)
    l2_error = np.multiply(dh2.T, out_h2 * (1 - out_h2))
    dW2 = np.dot(out_h1.T, l2_error) 
    db2 = np.sum(l2_error, axis = 0, keepdims=True)

    # Calculating change in weights and bias from the input layer to the first hidden layer
    dh1 = np.dot(w2, l2_error.T)
    l1_error = np.multiply(dh1.T, out_h1 * (1 - out_h1))
    dW1 = np.dot(x.T, l1_error) 
    db1 = np.sum(l1_error, axis=0, keepdims=True)
